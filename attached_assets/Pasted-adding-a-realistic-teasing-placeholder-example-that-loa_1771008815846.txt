Update your JavaScript to handle the sample data properly
Make sure calcNet() includes all the fields you're using now. Here's an expanded version that matches your screenshot (replace your existing calcNet() function):JavaScriptfunction calcNet() {
  // Get values – use ?. to avoid errors if element missing
  const price          = parseFloat(document.getElementById('sale-price')?.value) || 0;
  const mortgage       = parseFloat(document.getElementById('mortgage-balance')?.value) || 0;
  const commissionPct  = (parseFloat(document.getElementById('commission-rate')?.value) || 6.0) / 100;
  const grtPct         = (parseFloat(document.getElementById('nm-grt-rate')?.value) || 7.625) / 100;
  const survey         = parseFloat(document.getElementById('survey-ilr')?.value) || 275;
  const annualTaxes    = parseFloat(document.getElementById('annual-taxes')?.value) || 3000;
  const concessions    = parseFloat(document.getElementById('seller-concessions')?.value) || 0;
  const repairs        = parseFloat(document.getElementById('repairs')?.value) || 0;

  // Simple proration example: assume half year for sample (adjust logic later if needed)
  const taxProration   = annualTaxes / 2;   // seller share – you'll refine this with month picker

  const commissionCost = price * commissionPct;
  const grtCost        = commissionCost * grtPct;
  const titleEscrow    = price * 0.008;     // rough 0.8% estimate – replace with your real formula

  const totalDeductions = 
    mortgage + 
    commissionCost + 
    grtCost + 
    titleEscrow + 
    taxProration + 
    survey + 
    concessions + 
    repairs;

  const net = price - totalDeductions;

  return {
    net,
    sale: price,
    mortgage,
    commission: commissionCost,
    nmGrt: grtCost,
    titleEscrow,
    taxProration,
    survey,
    concessions,
    repairs
  };
}
